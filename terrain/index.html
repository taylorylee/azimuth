Cloudflare Worker — CORS Proxy

Deploy this Worker to enable Real DEM mode. Free tier handles thousands of requests/day.

// terrain-proxy/index.js
export default {
  async fetch(request) {
    const url = new URL(request.url);
    const target =
      'https://portal.opentopography.org'
      + url.pathname
      + url.search;
    const resp = await fetch(target, {
      headers: { Accept: '*/*' }
    });
    const headers = new Headers(resp.headers);
    headers.set('Access-Control-Allow-Origin', '*');
    return new Response(resp.body, {
      status: resp.status,
      headers
    });
  }
};

Deploy: wrangler deploy — paste your *.workers.dev URL into Location → Proxy URL.
